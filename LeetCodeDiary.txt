Задача 605.

	Моя идея. 
	1. Пометить незанятые участки (первый проход)
	2. Разместить цветы там где это возможно (учитываем, что каждый новый цветок исключает посадку в сосоедних с ним клетках) (второй проход)

		#include <iostream>
		#include <vector>

		class Solution 
		{
		public:
		    bool canPlaceFlowers(std::vector<int>& flowerbed, int n) 
		    {
		    	int index = 0; 
				int end = flowerbed.size();
				while(index < end)								// первый проход
				{
					if (flowerbed[index] == 1)
					{
						if (index != 0)
						{
							flowerbed[index - 1] = 1;  
						}
						if (index != end - 1)
						{
							flowerbed[index + 1] = 1;
						}
						index = index + 2;
						continue;
					}
					++index;
				}

				index = 0;
				int empty = 0;
				while(index < end)							// второй проход, размещаем цветы
				{
					if (flowerbed[index] == 0)
					{
						if (index != 0)
						{
							flowerbed[index - 1] = 1;
						}
						if (index != end - 1)
						{
							flowerbed[index + 1] = 1;
						}

						flowerbed[index] = 1;
						index = index + 2;
						++empty;
						continue;
					} 
					++index;
				}
				return empty >= n;
		    }
		};

	Лучшее решение от deep seek

		class Solution 
		{
		public:
		    bool canPlaceFlowers(vector<int>& flowerbed, int n) 
		    {
			int count = 0;
			int size = flowerbed.size();
			
			for (int i = 0; i < size && count < n; i++) 
			{
			    if (flowerbed[i] == 0) 						// посадка возможна только в свободных клетках
			    {
				// Проверяем, можно ли посадить здесь
				bool leftEmpty = (i == 0) || (flowerbed[i-1] == 0);		// есть ли ограничение слева ? + граничные условия
				bool rightEmpty = (i == size-1) || (flowerbed[i+1] == 0);	// есть ли ограничение справа ? + граничные условия
				
				if (leftEmpty && rightEmpty) 					// оба условия выполнены
				{
				    flowerbed[i] = 1; // сажаем
				    count++;
				    i++; // пропускаем следующую позицию (она теперь занята)	// ускоряем цикл на один шаг
				}
			    }
			}
			return count >= n;
		    }
		};

	Реверс слов в строке (Reverse Words in a String)
	Строка может иметь несколько пробелов между словами, их требуется исключить.

	Задача делится на несколько классических подзадач.

	151.1. Удаление пробелов
	Начнем с задачи исключения пробелов в строке. Она решается с использованием метода двух указателей.
	Первый указатель (это индекс j, назовем его forward) будет указывать на начало текста, 
	второй указатель (индекс i, назовем его back) указывает на свободное место, куда можно перенести слово.

	void reduceSpaces(std::string& s)
    {
	    int n = s.length();
	    int i = 0, j = 0;
	    while (j < n) 
	    {
			while (j < n && s[j] == ' ') j++;             	// forward пропускает пробелы, пока не встретит начало слова, возможно этот цикл сработает только один раз в начале
			while (j < n && s[j] != ' ') s[i++] = s[j++];	// переносим слово влево, остановка на пробеле, очень хороший ход
			while (j < n && s[j] == ' ') j++;				// forward пропускает пробелы
			if (j < n) s[i++] = ' ';						// back передвигается правее, на второй пробел после слова, в точности где должно начинаться слово.
	    }
	    s.resize(i);
    }
	   

    151.2. Реверс элементов массива
    Нужно переставить местами элемент a[0] с элементом a[n - 1], элемент a[1] с элементом a[n - 2] и вообще каждый элемент a[i] с элементом a[n - 1 - i].
    Индекс i растет, разность n - 1 - i убывает. Обмен необходимо производить пока i < n - 1 - i.

    	i < n - 1 - i  	(1)
    	2 * i < n - 1  	(2)

    	поскольку числа целые, легко увидеть, что самое большое удовлетворящее (2) это 2 * i = n - 2
    	значит справедливо равносильное неравенство

    	2 * i <= n - 2 	(3)

    	выполним целочисленное деление i <= (n - 2) / 2

    	i <= n/2 - 1 	(4)

    	и, наконец перейдем к строгому неравенству

    	i < n/2 		(5)

    	Это удобно, потому что n - это длина диапазона.
    	Приводим алгоритм для реверса выбранного участка строки.

    	void reverseStr(std::string& s, int begin, int end)
		{
			int n = end - begin;

			for (int i = 0; i < n/2; ++i)

Задача 238.

	Product of Array Except Self
	Произведение элементов массива.

	В каждую ячейку записать произведение элементов массива, за исключением элемента из данной ячейки. Обеспечить O(n).

	Первая мысль - найти общее произведение, произведние всего массива и для каждой ячейки находить частное
		answer[i] = product / a[i]
	Данный подход отбрасывается, поскольку возможно, что  a[i] = 0.

	Пусть у нас есть 2 дополнительных массива, в одном записаны префиксные произведения, во втором постфиксные произведения.
	Первый массив prefix
	a[0], a[0] * a[1], a[0] * a[1] * a[2], ..... , a[0], a[0] * a[1] * ..... * a[n - 2] * a[n - 1] - последний элемент, полное произведение
	Второй массив postfix
	a[0] * a[1] * ..... * a[n - 2] * a[n - 1], a[0] * a[1] * ..... * a[n - 2], .... , a[n - 2] * a[n - 1], a[n - 1]

	Для каждого элемента выходного массива справедливо равенство answer[i] = prefix[i] * postfix[i].

	А теперь, решение, где не потребуется дополнительная память, по совету deep seek.
		
		1. Заполняем массив answer префиксными произведениями.
				
			int prefix = 1;
			for (int i = 0; i < n; i++)
			{
				answer[i] = answer[i] * prefix;
				prefix = answer[i];
			}
	
		2. Накапливаем postfix, идем с конца массива nums[i]

			int postfix = 1;
			for (int j = n - 1; j > 0; j++)
			{
				answer[j] = postfix * answer[j - 1];
				postfix = postfix * nums[j];
			}
			answer[0] = postfix;

	    void moveZeroes(vector<int>& nums) 
	    {
	        int insertPos = 0; 		// позиция для вставки ненулевого элемента
	        
	        for (int num : nums)   	// Проходим по массиву 
	        {
	            if (num != 0) 
	            {
	                nums[insertPos++] = num;  	// соответственно при num = 0, insertPos остается на этом элементе 
	            }
	        }
	        
	        while (insertPos < nums.size()) 	// Заполняем оставшиеся позиции нулями
	        {
	            nums[insertPos++] = 0;
	        }
    	}

	она становится основной.
	
	    bool increasingTriplet(vector<int>& nums) 
    {
        auto n = nums.size();
        int i = 0;
        int A, B, a;

        while(i + 1 < n && nums[i] >= nums[i + 1]) { i++; }
        
        if (i + 1 < n)
        {
            A = nums[i];
            a = nums[i];
            B = nums[i + 1];
        }

        while(i + 1 < n)
        {
            ++i;
            if (nums[i] > B)
            {
                return true;
            }
            if (nums[i] > A)
            {
                B = nums[i];
            }
            else
            {
                if (nums[i] <= a)
                {
                    a = nums[i];
                }
                else
                {
                    A = a;
                    B = nums[i];
                }
            }
        }
        return false;
    }
    
    А теперь клссическое решение

    bool increasingTriplet(vector<int>& nums) 
    {
        int first = INT_MAX;
        int second = INT_MAX;
        
        for (int num : nums) 
        {
            if (num <= first) 
            {
                first = num;           // самый маленький кандидат
            } 
            else if (num <= second) 
            {
                second = num;          // второй кандидат
            } 
            else 
            {
                return true;           // нашли третий > second
            }
        }
        return false;
    }
